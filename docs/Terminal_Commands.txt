Commands for running microservices WITHOUT crashing VS Code

🎯 RECOMMENDED: Run microservices in separate Docker containers
This keeps VS Code lightweight! Services run independently. Works everywhere (local + Codespaces)!

# Build service images (first time only)
./services-control.sh build

# Start all services
./services-control.sh start

# Start individual services
./services-control.sh start accounts-service
./services-control.sh start products-service

# Check status and resource usage
./services-control.sh status

# View logs
./services-control.sh logs accounts-service

# Stop services
./services-control.sh stop

# Complete shutdown
./services-control.sh down

Benefits: ✅ No VS Code freezing ✅ Works on any machine ✅ Proper resource isolation

LEGACY (may freeze with 3+ services):
./service-manager.sh start  # Runs inside dev container

To kill frontend port

commands to stop all containers from running
# Stop all running containers
docker stop $(docker ps -aq)

# Remove all containers
docker rm $(docker ps -aq)

# Verify all are gone
docker ps -a


# To fix database not initializing in external container

# Convert line endings to Unix format
dos2unix init-database.sh || sed -i 's/\r$//' init-database.sh

# Now recreate the database from scratch
./services-control.sh down
docker volume rm gamestore_services_mysql-data
./services-control.sh start

# Wait 30 seconds for database to initialize
sleep 30

# Verify it worked
docker exec gamestore_services-db-1 mysql -u root -prootpassword Game_Store_System -e "SHOW TABLES;"


# 📊 Better formatted MySQL queries for readability

# Option 1: Use --table flag with column for perfect alignment
docker exec gamestore_services-db-1 mysql -u root -prootpassword Game_Store_System --table -e "SELECT * FROM Accounts;" | column -t -s $'\t'

# Option 2: Use less -S for horizontal scrolling on wide tables (recommended for many columns)
docker exec gamestore_services-db-1 mysql -u root -prootpassword Game_Store_System --table -e "SELECT * FROM Accounts;" | less -S

# Option 3: Format with specific columns for better control
docker exec gamestore_services-db-1 mysql -u root -prootpassword Game_Store_System --table -e "SELECT AccountID, AccountName, AccountEmail FROM Accounts;"

# Option 4: Pretty print with vertical bars (best for narrow terminals)
docker exec gamestore_services-db-1 mysql -u root -prootpassword Game_Store_System -e "SELECT * FROM Accounts;" | sed 's/\t/  |  /g'

# Show tables with better formatting
docker exec gamestore_services-db-1 mysql -u root -prootpassword Game_Store_System --table -e "SHOW TABLES;"

# Query Order_Items table (shows which products are in each order)
docker exec gamestore_services-db-1 mysql -u root -prootpassword Game_Store_System --table -e "SELECT * FROM Order_Items;"

# Query Order_Items with joined product info
docker exec gamestore_services-db-1 mysql -u root -prootpassword Game_Store_System --table -e "SELECT oi.*, p.GameTitle, p.AlbumTitle FROM Order_Items oi LEFT JOIN Products p ON oi.ProductID = p.ProductID;"

# Query Payments with Order_Items relationship
docker exec gamestore_services-db-1 mysql -u root -prootpassword Game_Store_System --table -e "SELECT p.PaymentID, p.OrderID, p.PaymentAmount, p.PaymentStatus, COUNT(oi.OrderItemID) as ItemCount FROM Payments p LEFT JOIN Order_Items oi ON p.OrderID = oi.OrderID GROUP BY p.PaymentID;"

# Interactive mode for best experience (allows scrolling and formatting)
docker exec -it gamestore_services-db-1 mysql -u root -prootpassword Game_Store_System

# Tips:
# - In less: Use arrow keys to scroll, 'q' to quit, Left/Right for horizontal scroll
# - --table flag enforces box-drawing characters around tables
# - For CSV export: Add --batch --skip-column-names flags
