do s=${svc%%/*}; line=$(grep -o '<counter type="LINE"[^/]*/>' "$svc" | tail -1); m=$(echo "$line" | 
grep -oP 'missed="\K[0-9]+'); c=$(echo "$line" | grep -oP 'covered="\K[0-9]+'); t=$((m+c)); 
printf "%-35s %3d%%  (%d/%d)\n" "$s" $((c*100/t)) "$c" "$t"; done

Commands for running microservices WITHOUT crashing VS Code

🎯 RECOMMENDED: Run microservices in separate Docker containers
This keeps VS Code lightweight! Services run independently. Works everywhere (local + Codespaces)!

# Build service images (first time only)
./services-control.sh build

# Start all services
./services-control.sh start

# Start individual services
./services-control.sh start accounts-service
./services-control.sh start products-service

# Check status and resource usage
./services-control.sh status

# View logs
./services-control.sh logs accounts-service

# Stop services
./services-control.sh stop

# Complete shutdown
./services-control.sh down

Benefits: ✅ No VS Code freezing ✅ Works on any machine ✅ Proper resource isolation

LEGACY (may freeze with 3+ services):
./service-manager.sh start  # Runs inside dev container

# 📊 Better formatted MySQL queries for readability

# Option 1: Use --table flag with column for perfect alignment
docker exec gamestore_services-db-1 mysql -u root -p"$MYSQL_ROOT_PASSWORD" Game_Store_System --table -e "SELECT * FROM Accounts;" | column -t -s $'\t'

# Option 2: Use less -S for horizontal scrolling on wide tables (recommended for many columns)
docker exec gamestore_services-db-1 mysql -u root -p"$MYSQL_ROOT_PASSWORD" Game_Store_System --table -e "SELECT * FROM Accounts;" | less -S

# Option 3: Format with specific columns for better control
docker exec gamestore_services-db-1 mysql -u root -p"$MYSQL_ROOT_PASSWORD" Game_Store_System --table -e "SELECT AccountID, AccountName, AccountEmail FROM Accounts;"

# Option 4: Pretty print with vertical bars (best for narrow terminals)
docker exec gamestore_services-db-1 mysql -u root -p"$MYSQL_ROOT_PASSWORD" Game_Store_System -e "SELECT * FROM Accounts;" | sed 's/\t/  |  /g'

# Show tables with better formatting
docker exec gamestore_services-db-1 mysql -u root -p"$MYSQL_ROOT_PASSWORD" Game_Store_System --table -e "SHOW TABLES;"

# Query Order_Items table (shows which products are in each order)
docker exec gamestore_services-db-1 mysql -u root -p"$MYSQL_ROOT_PASSWORD" Game_Store_System --table -e "SELECT * FROM Order_Items;"

# Query Order_Items with joined product info
docker exec gamestore_services-db-1 mysql -u root -p"$MYSQL_ROOT_PASSWORD" Game_Store_System --table -e "SELECT oi.*, p.GameTitle, p.AlbumTitle FROM Order_Items oi LEFT JOIN Products p ON oi.ProductID = p.ProductID;"

# Query Payments with Order_Items relationship
docker exec gamestore_services-db-1 mysql -u root -p"$MYSQL_ROOT_PASSWORD" Game_Store_System --table -e "SELECT p.PaymentID, p.OrderID, p.PaymentAmount, p.PaymentStatus, COUNT(oi.OrderItemID) as ItemCount FROM Payments p LEFT JOIN Order_Items oi ON p.OrderID = oi.OrderID GROUP BY p.PaymentID;"

# Interactive mode for best experience (allows scrolling and formatting)
docker exec -it gamestore_services-db-1 mysql -u root -p"$MYSQL_ROOT_PASSWORD" Game_Store_System

to find genre for all songs by genre
docker exec -i gamestore_services-db-1 mysql -u root -p Game_Store_System -e "
SELECT
  CASE WHEN ProductID > 0 THEN 'Library' ELSE 'iTunes' END AS Source,
  Genre,
  COUNT(*) AS Count
FROM AudioFeatures
GROUP BY Source, Genre
ORDER BY Source, Count DESC;"
